Guia de implementação de anúncios

Google Publisher Tag

ℹ️
Nota: Todo o script de configuração deve ser inserido uma única vez no <head> da página, antes de qualquer chamada a googletag.display().

Slots disponíveis

Esta implementação define cinco slots responsivos, cada um com size mappings adaptados ao viewport.

ID do slot Posição Desktop Mobile
ad-header Topo da página 970×250 · 970×90 · 728×90 320×100 · 320×50
ad-conteudo Meio do artigo 300×250 · 336×280 300×250 · 336×280
ad-sidebar Barra lateral 300×600 · 300×250 300×250
ad-mobile Exclusivo mobile — 320×100 · 320×50
ad-rodape Rodapé da página 970×90 · 728×90 320×100 · 320×50

1. Código do <head>

Cole este bloco inteiro dentro do <head> do seu HTML. Substitua {{CODE_NAME}} pelo identificador do seu site (ex: meusite).

Preenche {{MCM_CODE}} no ad unit
Preenche {{CODE_NAME}} nos trechos abaixo
Preenche {{PAGE_URL}} em pubads().set('page_url', …) 
<!-- Colar no <head> — uma única vez -->
<script async src="https://securepubads.g.doubleclick.net/tag/js/gpt.js" crossorigin="anonymous"></script>
<script>
window.googletag = window.googletag || { cmd: [] };

googletag.cmd.push(function () {

  // ── Size Mappings ──────────────────────────────
  var mapHeader = googletag.sizeMapping()
    .addSize([970, 0], [[970, 250], [970, 90], [728, 90]])
    .addSize([728, 0], [[728, 90]])
    .addSize([0,   0], [[320, 100], [300, 100], [320, 50], [300, 50]])
    .build();

  var mapLeaderboard = googletag.sizeMapping()
    .addSize([970, 0], [[970, 90], [728, 90]])
    .addSize([728, 0], [[728, 90]])
    .addSize([0,   0], [[320, 100], [300, 100], [320, 50], [300, 50]])
    .build();

  var mapSidebar = googletag.sizeMapping()
    .addSize([768, 0], [[300, 600], [300, 250]])
    .addSize([0,   0], [[300, 250]])
    .build();

  var mapRetangulo = googletag.sizeMapping()
    .addSize([0, 0], [[300, 250], [336, 280]])
    .build();

  var mapMobile = googletag.sizeMapping()
    .addSize([768, 0], [])
    .addSize([0,   0], [[320, 100], [300, 100], [320, 50], [300, 50]])
    .build();

  // ── Slots ──────────────────────────────────────
  googletag.defineSlot('/7542,{{MCM_CODE}}/parceiros/{{CODE_NAME}}',
    [[970, 250], [970, 90], [728, 90], [320, 100], [300, 100], [320, 50], [300, 50]],
    'ad-header')
    .defineSizeMapping(mapHeader).addService(googletag.pubads());

  googletag.defineSlot('/7542,{{MCM_CODE}}/parceiros/{{CODE_NAME}}',
    [[300, 250], [336, 280]],
    'ad-conteudo')
    .defineSizeMapping(mapRetangulo).addService(googletag.pubads());

  googletag.defineSlot('/7542,{{MCM_CODE}}/parceiros/{{CODE_NAME}}',
    [[300, 600], [300, 250]],
    'ad-sidebar')
    .defineSizeMapping(mapSidebar).addService(googletag.pubads());

  googletag.defineSlot('/7542,{{MCM_CODE}}/parceiros/{{CODE_NAME}}',
    [[728, 90], [970, 90], [320, 100], [300, 100], [320, 50], [300, 50]],
    'ad-rodape')
    .defineSizeMapping(mapLeaderboard).addService(googletag.pubads());

  googletag.defineSlot('/7542,{{MCM_CODE}}/parceiros/{{CODE_NAME}}',
    [[320, 100], [300, 100], [320, 50], [300, 50]],
    'ad-mobile')
    .defineSizeMapping(mapMobile).addService(googletag.pubads());

  googletag.pubads().enableSingleRequest();
  googletag.pubads().set('page_url', '{{PAGE_URL}}');
  googletag.pubads().collapseEmptyDivs();
  googletag.enableServices();
});
</script>
